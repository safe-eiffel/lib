feature -- Access

name:STRING

-- table, column, or procedure name

feature -- Element change

set_catalog (value:ECLI_VARCHAR)

-- set catalog wit value

require

value: value/=Void

ensure

void_if_null_value: value.is_nullimpliescatalog=Void
assigned_if_not_null: notvalue.is_nullimpliescatalog.is_equal(value.as_string)

set_name (value:ECLI_VARCHAR)

-- set name with value

require

value_not_void: value/=Void
value_not_null: notvalue.is_null

ensure

assigned: name.is_equal(value.as_string)

set_schema (value:ECLI_VARCHAR)

-- set schema with value

require

value: value/=Void

ensure

void_if_null_value: value.is_nullimpliesschema=Void
assigned_if_not_null: notvalue.is_nullimpliesschema.is_equal(value.as_string)
