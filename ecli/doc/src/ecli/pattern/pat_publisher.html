class PAT_PUBLISHER [G->PAT_SUBSCRIBER[PAT_PUBLISHER[G]]]

feature -- Initialization

default_create

feature -- Status report

count:INTEGER

-- count of subscribers

has_subscribed (subscriber:G):BOOLEAN

-- has this 'subscriber' subscribed to this service ?

require

valid_subscriber: subscriber/=Void

feature -- Element change

subscribe (subscriber:G)

-- subscribe statement 's'

require

valid_subscriber: subscriber/=Void
not_already_subscribed: nothas_subscribed(subscriber)

ensure

subscribed: has_subscribed(subscriber)
one_more: count=oldcount+1

feature -- Removal

unsubscribe (subscriber:G)

-- de-subscribe statement 's'

require

valid_subscriber: subscriber/=Void
subscribed: has_subscribed(subscriber)

ensure

not_subscribed: nothas_subscribed(subscriber)
one_less: count=oldcount-1

invariant

end
